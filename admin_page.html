    📖 API 对接文档

    以下接口供第三方开发者对接使用，需要先获取 Access Token。

    1. 获取数据列表

    GET /api/cddata/{project_id}?token={token}&queryType=0
    参数：project_id=项目ID, token=项目Token, queryType=0未读/1已读/-1全部

    2. 获取单条数据

    GET /api/cddata/{data_id}?token={token}
    返回该条数据的原始内容（纯文本）

    3. 上传数据

    POST /api/cddata/{project_id}?token={token}
    Body (JSON): {"数据名称": "xxx", "数据": "xxx"}

    4. 获取第一条未读数据（自动标已读）

    GET /api/cddata/{project_id}/fetchone?token={token}
    取该项目下第一条未读数据（按id升序），返回内容后自动标为已读。

    5. 标记数据状态

    POST /api/cddata/state/{data_id}?token={token}
    切换该条数据的已读/未读状态

    6. 删除第一条数据（即用即删）

    GET /api/cddata/{project_id}/popfirst?token={token}
    返回该项目下第一条未读数据（按id升序），返回后直接删除，适合队列式消息处理。

    7. 删除数据

    DELETE /api/cddata/{data_id}?token={token}

    获取 Token

    在云数据页面创建项目后，项目详情中会显示对应的 Token。
    Token 用于鉴权，请妥善保管。

    🤖 懒人精灵调用示例

    基于懒人精灵官方 httpGet / httpPost API (安卓端)

    -- ==============================
    -- 获取未读数据列表 (GET)
    -- queryType=0=未读, queryType=-1=全部
    -- ==============================

    local url = "https://qiezidata-production.up.railway.app/api/cddata/2?token=xxx&queryType=0"

    local ret, code = httpGet(url, 30)
    traceprint(ret)

    -- ==============================
    -- 上传数据 (POST)
    -- Body 字段: 数据名称 / 数据
    -- ==============================

    local url = "https://qiezidata-production.up.railway.app/api/cddata/2?token=xxx"
    local body = '{"数据名称":"test","数据":"hello"}'
    local headers = {}
    headers["Content-Type"] = "application/json"

    local ret, code = httpPost(url, body, 30, headers)
    traceprint(ret)

    -- ==============================
    -- 获取单条数据 (GET)
    -- {data_id} 替换为实际数据ID
    -- ==============================

    local url = "https://qiezidata-production.up.railway.app/api/cddata/123?token=xxx"

    local ret, code = httpGet(url, 30)
    traceprint(ret)

    -- ==============================
    -- 取第一条未读并标已读 (GET)
    -- ==============================

    local url = "https://qiezidata-production.up.railway.app/api/cddata/2/fetchone?token=xxx"

    local ret, code = httpGet(url, 30)
    traceprint(ret)

    -- ==============================
    -- 取第一条未读并删除 (队列模式) (GET)
    -- ==============================

    local url = "https://qiezidata-production.up.railway.app/api/cddata/2/popfirst?token=xxx"

    local ret, code = httpGet(url, 30)
    traceprint(ret)

    -- ==============================
    -- 处理 JSON 返回结果
    -- ==============================

    local ret, code = httpGet(url, 30)
    if code == 200 then
        local tb = jsonLib.decode(ret)
        traceprint("数据名称: " .. tb.data.数据名称)
        traceprint("数据内容: " .. tb.data.数据)
    end

    -- ==============================
    -- 异步请求 (不阻塞主线程)
    -- ==============================

    function callback(ret, code)
        traceprint("收到数据: " .. ret)
    end

    asynHttpGet(callback, url, 30)
